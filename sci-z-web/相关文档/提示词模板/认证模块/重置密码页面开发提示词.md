- **页面类型**：认证页面

- 极简设计：只有一个邮箱输入框

### 2. 核心功能实现
```javascript
<script setup>
import { ref, reactive, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { useI18n } from 'vue-i18n'
import { ElMessage } from 'element-plus'
import { BaseScrollbar } from '@/components/Common'
import { resetPassword } from '@/api/Auth'
import { validateEmail } from '@/utils/validate'
import { createLogger } from '@/utils/simpleLogger'

const router = useRouter()
const { t } = useI18n()
const logger = createLogger('ResetPassword')

// 响应式数据
const loading = ref(false)

// 表单数据
const resetForm = reactive({
  email: ''
})

// 验证状态
const emailError = ref('')
const emailSuccess = ref(false)

// 表单引用
const formRef = ref()

// 验证规则
const rules = {
  email: [
    { required: true, message: t('auth.resetPassword.emailRequired'), trigger: 'blur' },
    { type: 'email', message: t('auth.resetPassword.emailFormat'), trigger: 'blur' }
  ]
}

// 邮箱验证
const validateEmailField = () => {
  emailError.value = ''
  emailSuccess.value = false

  if (!resetForm.email) {
    emailError.value = t('auth.resetPassword.emailRequired')
    return false
  }

  if (!validateEmail(resetForm.email)) {
    emailError.value = t('auth.resetPassword.emailFormat')
    return false
  }

  // 模拟异步验证
  setTimeout(() => {
    emailSuccess.value = true
  }, 300)

  return true
}

// 重置密码处理
const handleReset = async () => {
  if (!formRef.value) return

  try {
    const valid = await formRef.value.validate()
    if (!valid) {
      logger.warn('Form validation failed')
      return
    }

    loading.value = true
    logger.info('Starting password reset', { email: resetForm.email })

    await resetPassword({ email: resetForm.email })
    ElMessage.success(t('auth.resetPassword.resetLinkSent'))
    logger.info('Password reset email sent successfully')

    // 重置表单
    resetForm.email = ''

    // 3秒后返回登录页面
    setTimeout(() => {
      logger.info('Auto redirecting to login page')
      router.push('/login')
    }, 3000)
  } catch (error) {
    logger.error('Password reset failed', error)
    ElMessage.error(t('auth.resetPassword.sendFailed'))
  } finally {
    loading.value = false
  }
}

// 返回登录
const handleBackToLogin = () => {
  router.push('/login')
}
</script>
```
