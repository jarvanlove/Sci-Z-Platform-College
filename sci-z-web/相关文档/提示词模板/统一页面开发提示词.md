- ✅ **列表页面**：需要搜索、表格、操作按钮

      </BaseButton>
    </div>
    
    <!-- 搜索区域 -->
    <BaseCard class="search-card">
      <SearchForm 
        :form-data="searchForm"
        @search="handleSearch"
        @reset="handleReset"
      />
    </BaseCard>
    
    <!-- 数据表格 -->
    <BaseCard class="table-card">
      <DataTable 
        :data="tableData"
        :loading="loading"
        :columns="tableColumns"
        @edit="handleEdit"
        @delete="handleDelete"
        @view="handleView"
      />
      
      <BasePagination 
        v-model:current-page="currentPage"

          :form-data="editFormData"
          @submit="handleSubmit"
          @cancel="editDialogVisible = false"
        />
      </BaseScrollbar>
    </BaseDialog>
  </div>
</template>

<script setup>
import { ref, reactive, computed, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { useI18n } from 'vue-i18n'
import { ElMessage, ElMessageBox } from 'element-plus'
import { Plus } from '@element-plus/icons-vue'

// 组件导入
import { BaseButton, BaseCard, BaseDialog, BasePagination, BaseScrollbar } from '@/components/Common'
import { SearchForm, DataTable } from '@/components/Business/List'
import { ProjectForm } from '@/components/Business/Project'

// API 导入
import { getProjectList, deleteProject, updateProject } from '@/api/Project'

// 状态管理
import { useAuthStore } from '@/store/modules/auth'

// 路由和国际化
const router = useRouter()
const { t } = useI18n()
const authStore = useAuthStore()

// 响应式数据
const loading = ref(false)
const tableData = ref([])
const total = ref(0)
const currentPage = ref(1)
const pageSize = ref(10)

// 搜索表单
const searchForm = reactive({
  name: '',
  status: '',
  department: '',
  dateRange: []
})

// 编辑弹窗
const editDialogVisible = ref(false)
const editFormData = ref({})
const isEdit = ref(false)

// 表格列配置
const tableColumns = computed(() => [
  { prop: 'name', label: t('project.list.columns.name'), width: 200 },
  { prop: 'department', label: t('project.list.columns.department'), width: 150 },
  { prop: 'status', label: t('project.list.columns.status'), width: 120 },
  { prop: 'progress', label: t('project.list.columns.progress'), width: 120 },
  { prop: 'createTime', label: t('project.list.columns.createTime'), width: 180 },
  { prop: 'actions', label: t('common.actions'), width: 200, fixed: 'right' }
])

// 编辑弹窗标题
const editDialogTitle = computed(() => 
  isEdit.value ? t('project.list.edit') : t('project.list.create')
)

// 获取项目列表
const fetchProjectList = async () => {
  try {
    loading.value = true
    const params = {
      page: currentPage.value,
      size: pageSize.value,
      ...searchForm
    }
    const response = await getProjectList(params)
    tableData.value = response.data.list
    total.value = response.data.total
  } catch (error) {
    ElMessage.error(t('common.fetchError'))
    console.error('Fetch project list error:', error)
  } finally {
    loading.value = false
  }
}

// 搜索处理
const handleSearch = (formData) => {
  Object.assign(searchForm, formData)
  currentPage.value = 1
  fetchProjectList()
}

// 重置搜索
const handleReset = () => {
  Object.keys(searchForm).forEach(key => {
    if (Array.isArray(searchForm[key])) {
      searchForm[key] = []
    } else {
      searchForm[key] = ''
    }
  })
  currentPage.value = 1
  fetchProjectList()
}

// 分页处理
const handlePageChange = () => {
  fetchProjectList()
}

// 创建项目
const handleCreate = () => {
  isEdit.value = false
  editFormData.value = {}
  editDialogVisible.value = true
}

// 编辑项目
const handleEdit = (row) => {
  isEdit.value = true
  editFormData.value = { ...row }
  editDialogVisible.value = true
}

// 查看项目
const handleView = (row) => {
  router.push(`/project/detail/${row.id}`)
}

// 删除项目
const handleDelete = async (row) => {
  try {
    await ElMessageBox.confirm(
      t('project.list.deleteConfirm'),
      t('common.confirm'),
      {
        confirmButtonText: t('common.confirm'),
        cancelButtonText: t('common.cancel'),
        type: 'warning'
      }
    )
    
    await deleteProject(row.id)
    ElMessage.success(t('common.deleteSuccess'))
    fetchProjectList()
  } catch (error) {
    if (error !== 'cancel') {
      ElMessage.error(t('common.deleteError'))
      console.error('Delete project error:', error)
    }
  }
}

// 提交表单
const handleSubmit = async (formData) => {
  try {
    if (isEdit.value) {
      await updateProject(editFormData.value.id, formData)
      ElMessage.success(t('common.updateSuccess'))
    } else {
      // 创建逻辑
      ElMessage.success(t('common.createSuccess'))
    }
    editDialogVisible.value = false
    fetchProjectList()
  } catch (error) {
    ElMessage.error(t('common.submitError'))
    console.error('Submit form error:', error)
  }
}

// 页面初始化
onMounted(() => {
  fetchProjectList()
})
</script>

<script setup>
import { ref, reactive, computed, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { useI18n } from 'vue-i18n'
import { ElMessage } from 'element-plus'
import { ArrowLeft } from '@element-plus/icons-vue'

// 组件导入
import { BaseButton, BaseCard } from '@/components/Common'
import { InfoCard, ProgressBar, Timeline, AttachmentList } from '@/components/Business/Detail'

// API 导入
import { getProjectDetail, getProjectTimeline, getProjectAttachments } from '@/api/Project'

// 路由和国际化
const route = useRoute()
const router = useRouter()
const { t } = useI18n()

// 响应式数据
const projectInfo = ref({})
const timelineData = ref([])
const attachmentData = ref([])
const timelineLoading = ref(false)
const attachmentLoading = ref(false)

// 信息卡片列配置
const infoColumns = computed(() => [
  { key: 'name', label: t('project.detail.name') },
  { key: 'department', label: t('project.detail.department') },
  { key: 'leader', label: t('project.detail.leader') },
  { key: 'status', label: t('project.detail.status') },
  { key: 'startDate', label: t('project.detail.startDate') },
  { key: 'endDate', label: t('project.detail.endDate') },
  { key: 'budget', label: t('project.detail.budget') },
  { key: 'description', label: t('project.detail.description') }
])

// 获取项目详情
const fetchProjectDetail = async () => {
  try {
    const projectId = route.params.id
    const response = await getProjectDetail(projectId)
    projectInfo.value = response.data
  } catch (error) {
    ElMessage.error(t('common.fetchError'))
    console.error('Fetch project detail error:', error)
  }
}

// 获取项目时间线
const fetchProjectTimeline = async () => {
  try {
    timelineLoading.value = true
    const projectId = route.params.id
    const response = await getProjectTimeline(projectId)
    timelineData.value = response.data
  } catch (error) {
    ElMessage.error(t('common.fetchError'))
    console.error('Fetch project timeline error:', error)
  } finally {
    timelineLoading.value = false
  }
}

// 获取项目附件
const fetchProjectAttachments = async () => {
  try {
    attachmentLoading.value = true
    const projectId = route.params.id
    const response = await getProjectAttachments(projectId)
    attachmentData.value = response.data
  } catch (error) {
    ElMessage.error(t('common.fetchError'))
    console.error('Fetch project attachments error:', error)
  } finally {
    attachmentLoading.value = false
  }
}

// 返回上一页
const handleBack = () => {
  router.go(-1)
}

// 编辑项目
const handleEdit = () => {
  router.push(`/project/edit/${route.params.id}`)
}

// 查看进度
const handleProgress = () => {
  router.push(`/project/progress/${route.params.id}`)
}

// 上传附件
const handleUpload = (file) => {
  // 上传逻辑
  ElMessage.success(t('common.uploadSuccess'))
}

// 下载附件
const handleDownload = (file) => {
  // 下载逻辑
  ElMessage.success(t('common.downloadSuccess'))
}

// 删除附件
const handleDeleteAttachment = (file) => {
  // 删除逻辑
  ElMessage.success(t('common.deleteSuccess'))
}

// 页面初始化
onMounted(() => {
  fetchProjectDetail()
  fetchProjectTimeline()
  fetchProjectAttachments()
})
</script>

<template>

<script setup>
import { ref, reactive, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { useI18n } from 'vue-i18n'
import { ElMessage } from 'element-plus'
import { ArrowLeft } from '@element-plus/icons-vue'

// 组件导入
import { BaseCard } from '@/components/Common'
import { FormSection, FileUpload, FormActions } from '@/components/Business/Form'

// API 导入
import { createProject } from '@/api/Project'

// 路由和国际化
const router = useRouter()
const { t } = useI18n()

// 响应式数据
const formRef = ref()
const submitLoading = ref(false)

// 表单数据
const formData = reactive({
  name: '',
  department: '',
  leader: '',
  budget: 0,
  description: '',
  startDate: '',
  endDate: '',
  files: []
})

// 院系选项
const departments = ref([
  { label: '计算机科学与技术学院', value: 'cs' },
  { label: '人工智能学院', value: 'ai' },
  { label: '数学与统计学院', value: 'math' },
  { label: '物理与电子工程学院', value: 'physics' },
  { label: '化学与材料科学学院', value: 'chemistry' },
  { label: '生物科学与技术学院', value: 'biology' },
  { label: '经济管理学院', value: 'economics' },
  { label: '文学院', value: 'literature' },
  { label: '外国语学院', value: 'foreign' },
  { label: '教育学院', value: 'education' }
])

// 表单验证规则
const formRules = {
  name: [
    { required: true, message: t('project.create.nameRequired'), trigger: 'blur' },
    { min: 2, max: 50, message: t('project.create.nameLength'), trigger: 'blur' }
  ],
  department: [
    { required: true, message: t('project.create.departmentRequired'), trigger: 'change' }
  ],
  leader: [
    { required: true, message: t('project.create.leaderRequired'), trigger: 'blur' }
  ],
  budget: [
    { required: true, message: t('project.create.budgetRequired'), trigger: 'blur' }
  ],
  startDate: [
    { required: true, message: t('project.create.startDateRequired'), trigger: 'change' }
  ],
  endDate: [
    { required: true, message: t('project.create.endDateRequired'), trigger: 'change' }
  ]
}

// 文件变化处理
const handleFileChange = (files) => {
  formData.files = files
}

// 提交表单
const handleSubmit = async () => {
  if (!formRef.value) return
  
  try {
    const valid = await formRef.value.validate()
    if (!valid) return
    
    submitLoading.value = true
    
    // 处理表单数据
    const submitData = {
      ...formData,
      startDate: formData.startDate ? new Date(formData.startDate).toISOString() : '',
      endDate: formData.endDate ? new Date(formData.endDate).toISOString() : ''
    }
    
    await createProject(submitData)
    ElMessage.success(t('common.createSuccess'))
    router.push('/project/list')
  } catch (error) {
    ElMessage.error(t('common.submitError'))
    console.error('Submit form error:', error)
  } finally {
    submitLoading.value = false
  }
}

// 取消操作
const handleCancel = () => {
  router.go(-1)
}

// 返回上一页
const handleBack = () => {
  router.go(-1)
}

// 页面初始化
onMounted(() => {
  // 初始化逻辑
})
</script>

## 🎯 使用指南

### 1. 如何选择组件
- **列表页面**：使用 `SearchForm` + `DataTable` + `BasePagination`
- **详情页面**：使用 `InfoCard` + `ProgressBar` + `Timeline` + `AttachmentList`
- **表单页面**：使用 `FormSection` + `FileUpload` + `FormActions`

### 2. 如何调用 API
```vue
<script setup>
// 1. 导入 API
import { getProjectList, createProject } from '@/api/Project'

// 2. 定义响应式数据
const loading = ref(false)
const data = ref([])

// 3. 定义 API 调用函数
const fetchData = async () => {
  try {
    loading.value = true
    const response = await getProjectList({ page: 1, size: 10 })
    data.value = response.data
  } catch (error) {
    ElMessage.error('获取数据失败')
  } finally {
    loading.value = false
  }
}

// 4. 在生命周期中调用
onMounted(() => {
  fetchData()
})
</script>
```
